fn get_normalized_illumination(tile_pos: &Vec3, light_pos: &Vec3, light_strength: &f32) -> f32 {
    if (tile_pos.x - light_pos.x).abs() > *light_strength
        || (tile_pos.y - light_pos.y).abs() > *light_strength
    {
        return 0.0;
    }

    let d: f32 = (f32::powf((tile_pos.x - light_pos.x).abs(), 2.0)
        + f32::powf((tile_pos.y - light_pos.y).abs(), 2.0))
    .sqrt();

    (1.0 - (d / TILE_SIZE / *light_strength)).clamp(0.0, 1.0)
}
